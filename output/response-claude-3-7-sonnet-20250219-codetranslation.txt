def find_max(n):
    """
    递归查找n个整数中的最大值
    
    参数:
        n: 整数数量
    返回:
        最大值
    """
    if n == 1:
        # 如果只有一个数，直接返回它
        a = int(input())
        return a
    else:
        # 递归找出前n-1个数中的最大值
        max_val = find_max(n - 1)
        # 读取第n个数
        a = int(input())
        # 返回前n-1个数的最大值和当前数中的较大者
        return a if a > max_val else max_val


def main():
    # 读取整数数量
    n = int(input())
    # 查找最大值
    max_val = find_max(n)
    # 输出最大值
    print(max_val)


if __name__ == "__main__":
    main()
